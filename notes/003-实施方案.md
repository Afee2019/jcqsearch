### 步骤 1.1：scanner 模块实现

**internal/scanner/scanner.go 核心设计：**

```go
type Scanner struct {
    pool      *pgxpool.Pool
    batchSize int
    conc      int
}

func (s *Scanner) ScanAll(ctx context.Context) error {
    // 1. 查询所有 enabled=true 的 scan_paths
    // 2. 查询所有 enabled=true 的 ignore_rules
    // 3. 记录扫描开始时间 scanStartTime
    // 4. 对每个 scan_path 启动 goroutine 并发扫描
    // 5. 扫描完成后清理过期记录
}

func (s *Scanner) scanPath(ctx context.Context, sp model.ScanPath, rules []model.IgnoreRule, scanTime time.Time) error {
    var batch []model.FileEntry

    err := filepath.WalkDir(sp.Path, func(path string, d fs.DirEntry, err error) error {
        if err != nil {
            // 跳过无权限的目录，记录日志但不中断
            return filepath.SkipDir  // 或 nil
        }

        // 检查忽略规则
        if d.IsDir() && shouldIgnoreDir(d.Name(), rules) {
            return filepath.SkipDir
        }
        if !d.IsDir() && shouldIgnoreFile(d.Name(), rules) {
            return nil
        }

        // 检查深度限制
        if sp.MaxDepth > 0 && depth(path, sp.Path) > sp.MaxDepth {
            if d.IsDir() {
                return filepath.SkipDir
            }
            return nil
        }

        // 收集文件信息
        info, err := d.Info()
        if err != nil {
            return nil  // 跳过无法获取信息的文件
        }

        entry := model.FileEntry{
            Path:       path,
            Dir:        filepath.Dir(path),
            Name:       d.Name(),
            Stem:       stemOf(d.Name()),
            Ext:        extOf(d.Name()),
            IsDir:      d.IsDir(),
            Size:       info.Size(),
            ModTime:    info.ModTime(),
            ScanPathID: sp.ID,
            ScannedAt:  scanTime,
        }

        batch = append(batch, entry)
        if len(batch) >= s.batchSize {
            if err := s.upsertBatch(ctx, batch); err != nil {
                return err
            }
            batch = batch[:0]
        }
        return nil
    })

    // 写入最后一批
    if len(batch) > 0 {
        if err := s.upsertBatch(ctx, batch); err != nil {
            return err
        }
    }

    // 清理已删除文件
    return s.cleanupDeleted(ctx, sp.ID, scanTime)
}
```

**关键实现细节：**

| 细节 | 处理方式 |
|------|---------|
| 无权限目录 | `filepath.SkipDir` 跳过，stderr 打印警告 |
| 符号链接 | `WalkDir` 默认不跟随符号链接，避免循环引用 |
| 文件名解析 | `stemOf`: `strings.TrimSuffix(name, filepath.Ext(name))`；`extOf`: `strings.ToLower(strings.TrimPrefix(filepath.Ext(name), "."))` |
| 批量写入 | 使用 pgx 的 `Batch` 或手动拼接多行 `VALUES` |
| 并发控制 | `errgroup.Group` + `SetLimit(concurrency)` |
| 进度显示 | 每批写入后输出进度（已扫描文件数 / 已写入条数） |

### 步骤 1.2：批量 UPSERT 实现

```go
func (s *Scanner) upsertBatch(ctx context.Context, entries []model.FileEntry) error {
    if len(entries) == 0 {
        return nil
    }

    // 构造批量 INSERT ... ON CONFLICT 语句
    var b strings.Builder
    b.WriteString(`INSERT INTO files (path, dir, name, stem, ext, is_dir, size, mod_time, scan_path_id, scanned_at) VALUES `)

    args := make([]any, 0, len(entries)*10)
    for i, e := range entries {
        if i > 0 {
            b.WriteString(",")
        }
        base := i * 10
        fmt.Fprintf(&b, "($%d,$%d,$%d,$%d,$%d,$%d,$%d,$%d,$%d,$%d)",
            base+1, base+2, base+3, base+4, base+5,
            base+6, base+7, base+8, base+9, base+10)
        args = append(args, e.Path, e.Dir, e.Name, e.Stem, e.Ext,
            e.IsDir, e.Size, e.ModTime, e.ScanPathID, e.ScannedAt)
    }

    b.WriteString(` ON CONFLICT (path) DO UPDATE SET
        dir=EXCLUDED.dir, name=EXCLUDED.name, stem=EXCLUDED.stem,
        ext=EXCLUDED.ext, is_dir=EXCLUDED.is_dir, size=EXCLUDED.size,
        mod_time=EXCLUDED.mod_time, scan_path_id=EXCLUDED.scan_path_id,
        scanned_at=EXCLUDED.scanned_at`)

    _, err := s.pool.Exec(ctx, b.String(), args...)
    return err
}
```

### 步骤 2.1：searcher 模块实现

**internal/searcher/searcher.go 核心设计：**

```go
type SearchParams struct {
    Keyword   string     // 搜索关键词
    Ext       string     // 扩展名筛选（如 "pdf"）
    Exts      []string   // 多扩展名（如 ["pdf", "docx"]）
    After     *time.Time // 修改时间起始
    Before    *time.Time // 修改时间截止
    Dir       string     // 限定目录前缀
    MinSize   int64      // 最小文件大小
    MaxSize   int64      // 最大文件大小
    DirsOnly  bool       // 仅目录
    FilesOnly bool       // 仅文件
    Limit     int        // 返回条数
}

type SearchResult struct {
    Entries    []model.FileEntry
    Total      int           // 本次返回条数
    Duration   time.Duration // 查询耗时
}

type Searcher struct {
    pool      *pgxpool.Pool
    threshold float64
}

func (s *Searcher) Search(ctx context.Context, params SearchParams) (*SearchResult, error) {
    start := time.Now()

    // 动态构造 SQL
    var conditions []string
    var args []any
    argIdx := 1

    // 关键词模糊匹配
    if params.Keyword != "" {
        conditions = append(conditions, fmt.Sprintf("stem %% $%d", argIdx))
        args = append(args, params.Keyword)
        argIdx++

        conditions = append(conditions, fmt.Sprintf("similarity(stem, $%d) >= $%d", argIdx-1, argIdx))
        args = append(args, s.threshold)
        argIdx++
    }

    // 扩展名
    if len(params.Exts) > 0 {
        conditions = append(conditions, fmt.Sprintf("ext = ANY($%d)", argIdx))
        args = append(args, params.Exts)
        argIdx++
    } else if params.Ext != "" {
        conditions = append(conditions, fmt.Sprintf("ext = $%d", argIdx))
        args = append(args, strings.ToLower(params.Ext))
        argIdx++
    }

    // 时间范围
    if params.After != nil {
        conditions = append(conditions, fmt.Sprintf("mod_time >= $%d", argIdx))
        args = append(args, *params.After)
        argIdx++
    }
    if params.Before != nil {
        conditions = append(conditions, fmt.Sprintf("mod_time < $%d", argIdx))
        args = append(args, *params.Before)
        argIdx++
    }

    // 目录限定
    if params.Dir != "" {
        conditions = append(conditions, fmt.Sprintf("dir LIKE $%d", argIdx))
        args = append(args, params.Dir+"%")
        argIdx++
    }

    // 文件大小
    if params.MinSize > 0 {
        conditions = append(conditions, fmt.Sprintf("size >= $%d", argIdx))
        args = append(args, params.MinSize)
        argIdx++
    }
    if params.MaxSize > 0 {
        conditions = append(conditions, fmt.Sprintf("size <= $%d", argIdx))
        args = append(args, params.MaxSize)
        argIdx++
    }

    // 文件/目录筛选
    if params.FilesOnly {
        conditions = append(conditions, "is_dir = false")
    } else if params.DirsOnly {
        conditions = append(conditions, "is_dir = true")
    }

    // 构造完整 SQL
    query := "SELECT path, dir, name, stem, ext, is_dir, size, mod_time FROM files"
    if len(conditions) > 0 {
        query += " WHERE " + strings.Join(conditions, " AND ")
    }

    // ORDER BY
    if params.Keyword != "" {
        query += fmt.Sprintf(" ORDER BY similarity(stem, $1) DESC, mod_time DESC")
    } else {
        query += " ORDER BY mod_time DESC"
    }

    // LIMIT
    query += fmt.Sprintf(" LIMIT $%d", argIdx)
    args = append(args, params.Limit)

    // 执行查询
    rows, err := s.pool.Query(ctx, query, args...)
    // ... scan rows into []model.FileEntry

    return &SearchResult{
        Entries:  entries,
        Total:    len(entries),
        Duration: time.Since(start),
    }, nil
}
```

### 步骤 2.4：时间与大小参数解析

```go
// 解析 --after/--before 参数
// 支持格式: "2024-01", "2024-01-15", "2024-01-15 10:30"
func parseTime(s string) (time.Time, error) {
    formats := []string{
        "2006-01-02 15:04",
        "2006-01-02",
        "2006-01",
    }
    for _, f := range formats {
        if t, err := time.ParseInLocation(f, s, time.Local); err == nil {
            return t, nil
        }
    }
    return time.Time{}, fmt.Errorf("无法解析时间: %s", s)
}

// 解析 --recent 参数
// 支持格式: "7d", "24h", "30m"
func parseRecent(s string) (time.Time, error) {
    if len(s) < 2 {
        return time.Time{}, fmt.Errorf("无效的时间: %s", s)
    }
    unit := s[len(s)-1]
    val, err := strconv.Atoi(s[:len(s)-1])
    if err != nil {
        return time.Time{}, err
    }
    now := time.Now()
    switch unit {
    case 'd':
        return now.AddDate(0, 0, -val), nil
    case 'h':
        return now.Add(-time.Duration(val) * time.Hour), nil
    case 'm':
        return now.Add(-time.Duration(val) * time.Minute), nil
    default:
        return time.Time{}, fmt.Errorf("不支持的时间单位: %c（支持 d/h/m）", unit)
    }
}

// 解析 --larger/--smaller 参数
// 支持格式: "100M", "1G", "500K", "1024"(字节)
func parseSize(s string) (int64, error) {
    s = strings.TrimSpace(s)
    if len(s) == 0 {
        return 0, nil
    }
    unit := strings.ToUpper(s[len(s)-1:])
    var multiplier int64 = 1
    numStr := s
    switch unit {
    case "K":
        multiplier = 1024
        numStr = s[:len(s)-1]
    case "M":
        multiplier = 1024 * 1024
        numStr = s[:len(s)-1]
    case "G":
        multiplier = 1024 * 1024 * 1024
        numStr = s[:len(s)-1]
    }
    val, err := strconv.ParseInt(numStr, 10, 64)
    if err != nil {
        return 0, fmt.Errorf("无法解析大小: %s", s)
    }
    return val * multiplier, nil
}
```

func highlightKeyword(name, keyword string) string {
    highlight := color.New(color.FgRed, color.Bold).SprintFunc()
    // 不区分大小写查找并替换为高亮版本
    lower := strings.ToLower(name)
    kwLower := strings.ToLower(keyword)
    idx := strings.Index(lower, kwLower)
    if idx < 0 {
        return name
    }
    return name[:idx] + highlight(name[idx:idx+len(keyword)]) + name[idx+len(keyword):]
}

## 九、风险与应对

| 风险 | 影响 | 应对方案 |
|------|------|---------|
| macOS 无 `zh_CN.UTF-8` locale | 数据库创建失败 | 检查 `locale -a`，必要时使用 `en_US.UTF-8`（pg_trgm 不依赖 locale） |
| pg_trgm 对短中文词效果差 | 单字搜索无结果 | 降低 similarity_threshold 或 fallback 到 `ILIKE` |
| 文件数过多（>100 万） | 扫描和搜索变慢 | 合理配置 scan_paths + ignore_rules 控制范围 |
| 符号链接循环 | 扫描死循环 | `WalkDir` 默认不跟随符号链接，安全 |
| 文件权限不足 | 扫描中断 | 跳过无权限目录/文件，记录日志 |
